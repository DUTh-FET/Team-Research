\newlength{\spacebox}

%%%%%%%%%%%%%%% Name/Title/Affiliation Style %%%%%%%%%%%%%%%%
\newcommand{\MyName}[1]{
		\Huge \usefont{OT1}{phv}{b}{n}\hfill \textit{#1}
		\par \normalsize \normalfont}
\newcommand{\MyTitle}[1]{
		\large \usefont{OT1}{phv}{m}{n}\hfill \textit{#1}
		\par \normalsize \normalfont}
\newcommand{\MyAffiliation}[1]{
		\large \usefont{OT1}{phv}{m}{n}\hfill \textit{#1}
		\par \normalsize \normalfont}

%%%%%%%%%%%%%%% Define Command "NewPart" for non-numbered uppercase section titles %%%%%%%%%%%%%%%%
\newcommand{\NewPart}[1]{\section*{\uppercase{#1}}}

%%%%%%%%%%%%%%% Define Personal Details Entry Style %%%%%%%%%%%%%%%%
\newcommand{\PersonalEntry}[2]{
		\noindent\hangindent=2em\hangafter=0 % Indentation
		\parbox{1.2\spacebox}{        % Box to align text
		\textit{#1}}		       % Entry name (birth, address, etc.)
		\hspace{1.5em} #2 \par}    % Entry value
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%% Define Skills Entry Style %%%%%%%%%%%%%%%%
\newcommand{\SkillsEntry}[2]{      % Same as \PersonalEntry
		\noindent\hangindent=2em\hangafter=0 % Indentation
		\parbox{1.2\spacebox}{        % Box to align text
		\textit{#1}}			   % Entry name (birth, address, etc.)
		\hspace{1.5em} #2 \par}    % Entry value
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%% Define Education Entry Style %%%%%%%%%%%%%%%%		
\newcommand{\EducationEntry}[4]{
		\noindent \textbf{#1} \hfill      % Study
		\colorbox{Black}{%
			\parbox{6em}{%
			\hfill\color{White}#2}} \par  % Duration
		\noindent \textit{#3} \par        % School
		\noindent\hangindent=2em\hangafter=0 \small #4 % Description
		\normalsize \par}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%% Define Honours/Awards Entry Style %%%%%%%%%%%%%%%%	
\newcommand{\AwardEntry}[3]{
		\noindent \textbf{#1} \hfill      % Honour/Award Title
		\colorbox{Black}{\color{White}#2} \par  % Date
		\noindent \textit{#3} \par              % Description
		\normalsize \par}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%% Define Work Experience Entry Style %%%%%%%%%%%%%%%%	
\newcommand{\WorkEntry}[4]{				  % Same as \EducationEntry
		\noindent \textbf{#1} \hfill      % Jobname
		\colorbox{Black}{\color{White}#2} \par  % Duration
		\noindent \textit{#3} \par              % Company
		\noindent\hangindent=2em\hangafter=0 \small #4 % Description
		\normalsize \par}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%% Define Miscellaneous Type Entry Style %%%%%%%%%%%%%%%%	
\newcommand{\MiscEntry}[2]{
		\noindent \textbf{#1} \par      % Title
		\noindent \textit{#2} \par              % Description
		\normalsize \par}
\newcommand{\MiscEntryDate}[3]{
		\noindent \textbf{#1} \hfill      % Title
		\colorbox{Black}{\color{White}#2} \par  % Date
		\noindent \textit{#3} \par              % Description
		\normalsize \par}

\NewPart{Publications}

%%%%%%% Here you can generate your publication list automatically by including them 
%%%%%%% in the corresponding ".bib" file. This method generates the whole content of each ".bib".
%%%%%%% Also, you can easily add new categories, like "Book Chapters" by
%%%%%%% coping the content (5 lines) of the "\item".
\begin{enumerate}[label=\textbf{\Alph*.},leftmargin=*]
    \item   \textbf{Referred Journal Articles}
            \begin{refsection}[Files/journals.bib]
            \nocite{*}
            \printbibliography[heading=none]
            \end{refsection}
    \item   \textbf{Conference Presentations}
            \begin{refsection}[Files/conferences.bib]
            \nocite{*}
            \printbibliography[heading=none]
            \end{refsection}
\end{enumerate}
